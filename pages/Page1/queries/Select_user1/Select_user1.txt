SELECT
  u.*,
  v.is_running_exp,
  v.guide_name,
  g.is_guide_exp,
  g.guiding_pace,
  g.vi_count,
  g.vi_name,
  g.vi_record
FROM
  user u
LEFT JOIN vi v ON u.private_id = v.private_id AND u.type = 'VI'
LEFT JOIN guide g ON u.private_id = g.private_id AND u.type = 'GUIDE'
WHERE
  u.name LIKE '%{{user_tabel.searchText}}%'
ORDER BY
  u.{{ user_tabel.sortOrder.column || 'created_at' }}
  {{
user_tabel.sortOrder.column
? (user_tabel.sortOrder.order === "desc" ? "DESC" : "ASC")
: "DESC"
}}
LIMIT
  {{user_tabel.pageSize}}
OFFSET
  {{user_tabel.pageOffset}}
